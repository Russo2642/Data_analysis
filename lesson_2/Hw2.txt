# Программа выводить в консоль текст загадки и ждать ввода пользователя
# Программа после ввода пользователя ответа должна вывести в консоль результат: правильный ли ответ дал пользователь
# Загадок должно быть 3, тематика вопросов должна быть по первому занятию
# Дополнительные требования :
# Программа должна в конце игры сказать, сколько ответов дал пользователь: сколько из них было верных
# Программа должна не учитывать регистр ответа: "Python" и "python" оба должны быть правильным ответом на вопрос "Какой язык мы учим?"


print('Играем в загадки')
first_q = input('Загадка 1 \nКакой язык мы изучаем?:  ')
first_a = 'python'
if first_a in first_q.lower():
    print('Правильно')
    a = 1
else:
    print('Неправильно')
print('Идём дальше')
second_q = input('Загадка 2 \nTrue+True= ?')
second_a = 'true'
if second_a == second_q.lower():
    print('Правильно')
    a = a + 1
else:
    print('Неправильно')
print('Идём дальше')
third_q = input('Загадка 3 \n Data type for numbers ?')
third_a = 'int'
if third_a == third_q.lower():
    print('Правильно')
    a = a + 1
else:
    print('Неправильно')
print(' Правильных ответов - ',a)
